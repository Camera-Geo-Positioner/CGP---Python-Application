pressed: bool = False

    return mouseX, mouseY


def MouseClicked():
    """
    Check if the mouse has been clicked.

    Returns
    -------
    bool
    """
    return mouseClicked


def SetMouseClicked(value: bool):
    """
    Set the mouseClicked.

    Parameters
    ----------
    value: Set mouseClicked to the value
    """
    global mouseClicked
    mouseClicked = value


def MouseClick(event, x, y, flags, param):
    """
    Event handler for mouse click events.

    Parameters
    ----------
    event: The mouse event that triggerd the event handler
    x: The x position of the mouse
    y: The y position of the mouse
    """
    global mouseX, mouseY
    global pressed
    global mouseClicked

    if event == cv2.EVENT_LBUTTONDOWN and not pressed:
        pressed = True
        mouseX, mouseY = x, y
        mouseClicked = True
    if event == cv2.EVENT_LBUTTONUP and pressed:
        pressed = False


class CameraWindow:
    """CameraWindow object, controls an openCV window."""
    windowName: str = ''
    targetWidth: int = 960
    targetHeight: int = 540
    windowOpen: bool = False

    def __init__(self, windowName: str, targetWidth=960, targetHeight=540):
        """
        Constructor of the CameraWindow.

        Parameters
        ----------
        windowName: The name that is used for the window
        targetWidth: The width at which the window should render frames
        targetHeight: The height at which the window should render frames
        """
        self.windowName = ''
        if windowName is not None:
            self.windowName = windowName
        # Ensure a minimum size of 1x1
        self.targetWidth = max(targetWidth, 1)
        self.targetHeight = max(targetHeight, 1)

    def OpenWindow(self):  # pragma: no cover
        """Open the OpenCV window."""
        cv2.namedWindow(self.windowName)
        cv2.setMouseCallback(self.windowName, MouseClick)
        self.windowOpen = True

    def CloseWindow(self):  # pragma: no cover
        """Close the OpenCV window."""
        if self.windowOpen:
            cv2.destroyWindow(self.windowName)
        self.windowOpen = False

    def ResizeFrame(self, frame):
        """
        Resize a frame to the window size

        Parameters
        ----------
        frame: The frame to be resized

        Returns
        -------
        Frame
        """
        return cv2.resize(frame, (self.targetWidth, self.targetHeight), interpolation=cv2.INTER_LINEAR)

    def ShowFrame(self, frame):  # pragma: no cover
        """
        Display a frame in the window, the frame is first resized to the window size.

        Parameters
        ----------
        frame: The frame to be displayed
        """
        if not self.windowOpen:
            return
        frameResized = self.ResizeFrame(frame)
        cv2.imshow(self.windowName, frameResized)
