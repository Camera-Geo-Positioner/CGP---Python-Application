import pygame
from Windowing.IWindow import IWindow
from Windowing.Rendering import Rendering


class CameraWindow(IWindow):
    """ This class represents a camera window. """

    frame = None

    pressed = False
    mouseX = 0
    mouseY = 0

    def __init__(self, windowName: str, targetWidth=960, targetHeight=540):
        """ Constructor a new camera window instance.

        Parameters
        ----------
        windowName : str
            The name of the window.
        targetWidth : int
            The target width of the window.
        targetHeight : int
            The target height of the window.
        """

        # Ensure a minimum size of 1x1
        targetWidth = max(targetWidth, 1)
        targetHeight = max(targetHeight, 1)

        # Call super
        super(CameraWindow, self).__init__(targetWidth, targetHeight, windowName)

        # Set mouse callback
        self.SetMouseCallback(self.MouseClick)

    def Update(self):
        """ Updates the camera window. """

        super(CameraWindow, self).Update()
        pass

    def Render(self, surface):  # pragma: no cover
        """ Renders the camera window.

        Parameters
        ----------
        surface : pygame.Surface
            The surface to render to.
        """

        # If there is currently no frame, return
        if self.frame is None:
            return

        # Calculate aspect ratio of image
        aspectRatio = self.frame.get_width() / self.frame.get_height()

        # Draw scaled image
        if self.width / self.height <= aspectRatio:
            # Calculate adjusted height
            adjustedHeight = self.width / aspectRatio

            # Render the image to the surface at the center of the screen,
            # with the adjusted height
            surface.blit(
                pygame.transform.scale(self.frame, (self.width, adjustedHeight)),
                (0, self.height / 2 - adjustedHeight / 2)
            )
        else:
            # Calculate adjusted width
            adjustedWidth = self.height * aspectRatio

            # Render the image to the surface at the center of the screen,
            # with the adjusted width
            surface.blit(
                pygame.transform.scale(self.frame, (adjustedWidth, self.height)),
                (self.width / 2 - adjustedWidth / 2, 0)
            )

        # Draw cross, if mouse was clicked
        if self.pressed:
            Rendering.RenderLine(surface, self.mouseX - 4, self.mouseY, self.mouseX + 4, self.mouseY, (255, 0, 0))
            Rendering.RenderLine(surface, self.mouseX, self.mouseY - 4, self.mouseX, self.mouseY + 4, (255, 0, 0))

        # Call super
        super(CameraWindow, self).Render(surface)
        pass

    def MousePosition(self):
        """ Get the last mouse position

        Returns
        -------
        (int, int)
            The last mouse position.
        """

        return self.mouseX, self.mouseY

    def AbsoluteMousePosition(self):
        """ Get the absolute mouse position

        Returns
        -------
        (int, int)
            The absolute mouse position.
        """

        # If there is no current frame or the
        # window size is invalid, return the
        # regular mouse position
        if self.frame is None or self.width <= 0 or self.height <= 0:
            return self.MousePosition()

        return \
            (self.mouseX / self.width) * self.frame.get_width(), \
            (self.mouseY / self.height) * self.frame.get_height()

    def MouseClick(self, event, x, y):
        """ Handle a mouse click.

        Parameters
        ----------
        event : int
            The event type.
        x : int
            The x position of the mouse.
        y : int
            The y position of the mouse.
        """

        if event == pygame.MOUSEBUTTONDOWN:
            self.pressed = True
            self.mouseX, self.mouseY = x, y

    def ShowFrame(self, frame):
        """ Show a frame in the window.

        Parameters
        ----------
        frame : pygame.Surface or cv2.Image
            The frame to show.
        """

        # Set image property;
        # Check if the image is an OpenCV image
        if isinstance(frame, pygame.Surface):
            self.frame = frame
        else:
            self.frame = CameraWindow.ConvertOpenCVImageToPyGameImage(frame)

        # Set invalidated
        self.invalidated = True

    @staticmethod
    def ConvertOpenCVImageToPyGameImage(image):
        """ Convert OpenCV images for Pygame.
            See https://blanktar.jp/blog/2016/01/pygame-draw-opencv-image.html

        Parameters
        ----------
        image : cv2.Image
            The image to convert.

        Returns
        -------
        pygame.Surface
            The converted image.
        """

        opencvImage = image[:, :, ::-1]  # Since OpenCV is BGR and pygame is RGB, it is necessary to convert it.
        shape = opencvImage.shape[
            1::-1]  # OpenCV(height,width,Number of colors), Pygame(width, height) So this is also converted.
        pygameImage = pygame.image.frombuffer(opencvImage.tobytes(), shape, 'RGB')

        return pygameImage
